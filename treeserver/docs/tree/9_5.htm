/* 年龄层次统计 */

PREDICATES

LONG fxr[10,1]

FROM \TG\DATA\SUB0.DBF

CONDITION

BEGIN

DEL<>1

BEGIN

STATISTICS fxr

#Y1: AGE(BIRTHDAY)<="25" /*25岁以上人数*/
#Y2: AGE(BIRTHDAY)<="30"&&AGE(BIRTHDAY)>"26" /*26岁至30岁人数*/
#Y3: AGE(BIRTHDAY)<="35"&&AGE(BIRTHDAY)>"31" /*31岁至35岁人数*/
#Y4: AGE(BIRTHDAY)<="40"&&AGE(BIRTHDAY)>"36" /*36岁至40岁人数*/
#Y5: AGE(BIRTHDAY)<="45"&&AGE(BIRTHDAY)>"41" /*41岁至45岁人数*/
#Y6: AGE(BIRTHDAY)<="50"&&AGE(BIRTHDAY)>"46" /*46岁至50岁人数*/
#Y7: AGE(BIRTHDAY)<="55"&&AGE(BIRTHDAY)>"51" /*51岁至55岁人数*/
#Y8: AGE(BIRTHDAY)<="60"&&AGE(BIRTHDAY)>"56" /*56岁至60岁人数*/
#Y9: AGE(BIRTHDAY)>"61" /*61岁以上人数*/
#Y10: AGE(BIRTHDAY)>"0" /*总人数*/
#X1:

END

END

ACTION WARRAY("\\TG\\FX\\NLH.DBF",fxr)
